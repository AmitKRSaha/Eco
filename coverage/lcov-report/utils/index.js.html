// About 1.5x faster than the two-arg version of Array#splice()
exports.spliceOne = function spliceOne(list, index) {
  for (let i = index, k = i + 1, n = list.length; k < n; i += 1, k += 1) {
    list[i] = list[k];
  }
  list.pop();
};
